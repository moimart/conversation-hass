14// rustdoc-stripper-ignore-next
15/// Trait containing manually implemented methods of
16/// [`DrawingArea`](crate::DrawingArea).
17pub trait DrawingAreaExtManual: sealed::Sealed + IsA<DrawingArea> + 'static {
18    #[doc(alias = "gtk_drawing_area_set_draw_func")]
19    #[doc(alias = "set_draw_func")]
20    fn unset_draw_func(&self) {
21        unsafe {
22            ffi::gtk_drawing_area_set_draw_func(
23                self.as_ref().to_glib_none().0,
24                None,
25                ptr::null_mut(),
26                None,
27            )
28        }
29    }
30
31    #[doc(alias = "gtk_drawing_area_set_draw_func")]
32    fn set_draw_func<P: FnMut(&DrawingArea, &cairo::Context, i32, i32) + 'static>(
33        &self,
34        draw_func: P,
35    ) {
36        unsafe extern "C" fn draw_func_func<
37            P: FnMut(&DrawingArea, &cairo::Context, i32, i32) + 'static,
38        >(
39            drawing_area: *mut ffi::GtkDrawingArea,
40            cr: *mut cairo::ffi::cairo_t,
41            width: libc::c_int,
42            height: libc::c_int,
43            user_data: glib::ffi::gpointer,
44        ) {
45            let drawing_area = from_glib_borrow(drawing_area);
46            let cr = from_glib_borrow(cr);
47            let callback: &RefCell<P> = &*(user_data as *mut _);
48            (callback.borrow_mut())(&drawing_area, &cr, width, height);
49        }
50
51        unsafe extern "C" fn destroy_func<
52            P: FnMut(&DrawingArea, &cairo::Context, i32, i32) + 'static,
53        >(
54            data: glib::ffi::gpointer,
55        ) {
56            let _callback: Box<RefCell<P>> = Box::from_raw(data as *mut _);
57        }
58
59        let callback: Box<RefCell<P>> = Box::new(RefCell::new(draw_func));
60        unsafe {
61            ffi::gtk_drawing_area_set_draw_func(
62                self.as_ref().to_glib_none().0,
63                Some(draw_func_func::<P> as _),
64                Box::into_raw(callback) as *mut _,
65                Some(destroy_func::<P> as _),
66            );
67        }
68    }
69}
70impl<O: IsA<DrawingArea>> DrawingAreaExtManual for O {}
