16/// Trait containing manually implemented methods of
17/// [`Accessible`](crate::Accessible).
18///
19/// ```no_run
20/// # use gtk4 as gtk;
21/// # use gtk::prelude::*;
22/// let entry = gtk::Entry::new();
23/// let label = gtk::Label::new(Some("Entry"));
24/// entry.update_property(&[
25///     gtk::accessible::Property::Description("Test"),
26///     gtk::accessible::Property::Orientation(gtk::Orientation::Horizontal),
27/// ]);
28/// entry.update_relation(&[gtk::accessible::Relation::LabelledBy(&[label.upcast_ref()])]);
29/// entry.update_state(&[gtk::accessible::State::Invalid(
30///     gtk::AccessibleInvalidState::Grammar,
31/// )]);
32/// ```
33pub trait AccessibleExtManual: sealed::Sealed + IsA<Accessible> {
34    #[doc(alias = "gtk_accessible_update_property")]
35    #[doc(alias = "gtk_accessible_update_property_value")]
36    fn update_property(&self, properties: &[Property]) {
37        let mut properties_ptr = vec![];
38        let mut values = vec![];
39        for prop in properties {
40            let (p, v) = prop.to_property_value();
41            properties_ptr.push(p.into_glib());
42            values.push(v);
43        }
44
45        unsafe {
46            ffi::gtk_accessible_update_property_value(
47                self.as_ref().to_glib_none().0,
48                properties.len() as i32,
49                mut_override(properties_ptr.as_ptr()),
50                ToGlibContainerFromSlice::to_glib_none_from_slice(&values).0,
51            )
52        }
53    }
54
55    #[doc(alias = "gtk_accessible_update_relation")]
56    #[doc(alias = "gtk_accessible_update_relation_value")]
57    fn update_relation(&self, relations: &[Relation]) {
58        let mut relations_ptr = vec![];
59        let mut values = vec![];
60        for relation in relations {
61            let (r, v) = relation.to_relation_value();
62            relations_ptr.push(r.into_glib());
63            values.push(v);
64        }
65
66        unsafe {
67            ffi::gtk_accessible_update_relation_value(
68                self.as_ref().to_glib_none().0,
69                relations.len() as i32,
70                mut_override(relations_ptr.as_ptr()),
71                ToGlibContainerFromSlice::to_glib_none_from_slice(&values).0,
72            )
73        }
74    }
75
76    #[doc(alias = "gtk_accessible_update_state")]
77    #[doc(alias = "gtk_accessible_update_state_value")]
78    fn update_state(&self, states: &[State]) {
79        let mut states_ptr = vec![];
80        let mut values = vec![];
81        for state in states {
82            let (s, v) = state.to_state_value();
83            states_ptr.push(s.into_glib());
84            values.push(v);
85        }
86
87        unsafe {
88            ffi::gtk_accessible_update_state_value(
89                self.as_ref().to_glib_none().0,
90                states.len() as i32,
91                mut_override(states_ptr.as_ptr()),
92                ToGlibContainerFromSlice::to_glib_none_from_slice(&values).0,
93            )
94        }
95    }
96}

133            Description(v) => (AccessibleProperty::Description, v.to_value()),
134            HasPopup(v) => (AccessibleProperty::HasPopup, v.to_value()),
135            KeyShortcuts(v) => (AccessibleProperty::KeyShortcuts, v.to_value()),
136            Label(v) => (AccessibleProperty::Label, v.to_value()),
137            Level(v) => (AccessibleProperty::Level, v.to_value()),
138            Modal(v) => (AccessibleProperty::Modal, v.to_value()),
139            MultiLine(v) => (AccessibleProperty::MultiLine, v.to_value()),
140            MultiSelectable(v) => (AccessibleProperty::MultiSelectable, v.to_value()),
141            Orientation(v) => (AccessibleProperty::Orientation, v.into_glib().to_value()),
142            Placeholder(v) => (AccessibleProperty::Placeholder, v.to_value()),
143            ReadOnly(v) => (AccessibleProperty::ReadOnly, v.to_value()),
144            Required(v) => (AccessibleProperty::Required, v.to_value()),
145            RoleDescription(v) => (AccessibleProperty::RoleDescription, v.to_value()),
146            Sort(v) => (AccessibleProperty::Sort, v.into_glib().to_value()),

156/// Type-safe enum container for
157/// [`AccessibleRelation`](crate::AccessibleRelation) values.
158#[derive(Debug)]
159#[non_exhaustive]
160pub enum Relation<'r> {
161    ActiveDescendant(&'r Accessible),
162    ColCount(i32),
163    ColIndex(i32),
164    ColIndexText(&'r str),
165    ColSpan(i32),
166    Controls(&'r [&'r Accessible]),
167    DescribedBy(&'r [&'r Accessible]),
168    Details(&'r [&'r Accessible]),
169    ErrorMessage(&'r [&'r Accessible]),
170    FlowTo(&'r [&'r Accessible]),
171    LabelledBy(&'r [&'r Accessible]),
172    Owns(&'r [&'r Accessible]),
173    PosInSet(i32),
174    RowCount(i32),
175    RowIndex(i32),
176    RowIndexText(&'r str),
177    RowSpan(i32),
178    SetSize(i32),
179}
180
181impl Relation<'_> {
182    fn to_relation_value(&self) -> (AccessibleRelation, Value) {
183        use Relation::*;
184
185        fn to_ref_list_value(objects: &[&Accessible]) -> Value {
186            skip_assert_initialized!();
187            let mut value = Value::from_type(glib::Type::POINTER);
188            let list =
189                ToGlibContainerFromSlice::<*mut glib::ffi::GList>::to_glib_container_from_slice(
190                    objects,
191                );
192            unsafe {
193                glib::gobject_ffi::g_value_set_pointer(
194                    value.to_glib_none_mut().0,
195                    list.0 as *mut std::ffi::c_void,
196                );
197            }
198            value
199        }
200
201        match self {
202            ActiveDescendant(v) => (AccessibleRelation::ActiveDescendant, v.to_value()),
203            ColCount(v) => (AccessibleRelation::ColCount, v.to_value()),
204            ColIndex(v) => (AccessibleRelation::ColIndex, v.to_value()),
205            ColIndexText(v) => (AccessibleRelation::ColIndexText, v.to_value()),
206            ColSpan(v) => (AccessibleRelation::ColSpan, v.to_value()),
207            Controls(v) => (AccessibleRelation::Controls, to_ref_list_value(v)),
208            DescribedBy(v) => (AccessibleRelation::DescribedBy, to_ref_list_value(v)),
209            Details(v) => (AccessibleRelation::Details, to_ref_list_value(v)),
210            ErrorMessage(v) => (AccessibleRelation::ErrorMessage, to_ref_list_value(v)),
211            FlowTo(v) => (AccessibleRelation::FlowTo, to_ref_list_value(v)),
212            LabelledBy(v) => (AccessibleRelation::LabelledBy, to_ref_list_value(v)),
213            Owns(v) => (AccessibleRelation::Owns, to_ref_list_value(v)),
214            PosInSet(v) => (AccessibleRelation::PosInSet, v.to_value()),
215            RowCount(v) => (AccessibleRelation::RowCount, v.to_value()),
216            RowIndex(v) => (AccessibleRelation::RowIndex, v.to_value()),
217            RowIndexText(v) => (AccessibleRelation::RowIndexText, v.to_value()),
218            RowSpan(v) => (AccessibleRelation::RowSpan, v.to_value()),
219            SetSize(v) => (AccessibleRelation::SetSize, v.to_value()),
220        }
221    }
222}

225/// Type-safe enum container for [`AccessibleState`](crate::AccessibleState)
226/// values.
227#[derive(Debug)]
228#[non_exhaustive]
229pub enum State {
230    Busy(bool),
231    Checked(AccessibleTristate),
232    Disabled(bool),
233    Expanded(Option<bool>),
234    Hidden(bool),
235    Invalid(AccessibleInvalidState),
236    Pressed(AccessibleTristate),
237    Selected(Option<bool>),
238}
239
240impl State {
241    fn to_state_value(&self) -> (AccessibleState, Value) {
242        use State::*;
243
244        fn to_optional_bool_value(b: &Option<bool>) -> Value {
245            skip_assert_initialized!();
246            b.map(|b| b as i32).unwrap_or(-1).to_value()
247        }
248
249        match self {
250            Busy(v) => (AccessibleState::Busy, v.to_value()),
251            Checked(v) => (AccessibleState::Checked, v.into_glib().to_value()),
252            Disabled(v) => (AccessibleState::Disabled, v.to_value()),
253            Expanded(v) => (AccessibleState::Expanded, to_optional_bool_value(v)),
254            Hidden(v) => (AccessibleState::Hidden, v.to_value()),
255            Invalid(v) => (AccessibleState::Invalid, v.into_glib().to_value()),
256            Pressed(v) => (AccessibleState::Pressed, v.into_glib().to_value()),
257            Selected(v) => (AccessibleState::Selected, to_optional_bool_value(v)),
258        }
259    }
260}
