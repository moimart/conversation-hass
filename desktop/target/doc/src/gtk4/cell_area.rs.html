7mod sealed {
8    pub trait Sealed {}
9    impl<T: super::IsA<super::CellArea>> Sealed for T {}

12// rustdoc-stripper-ignore-next
13/// Trait containing manually implemented methods of
14/// [`CellArea`](crate::CellArea).
15#[cfg_attr(feature = "v4_10", deprecated = "Since 4.10")]
16#[allow(deprecated)]
17pub trait CellAreaExtManual: sealed::Sealed + IsA<CellArea> {
18    #[doc(alias = "gtk_cell_area_add_with_properties")]
19    fn add_with_properties(
20        &self,
21        renderer: &impl IsA<CellRenderer>,
22        properties: &[(&str, &dyn ToValue)],
23    ) {
24        self.as_ref().add(renderer);
25        properties.iter().for_each(|(property_name, value)| {
26            self.cell_set(renderer, *property_name, *value);
27        });
28    }
29
30    #[doc(alias = "gtk_cell_area_cell_get_valist")]
31    #[doc(alias = "gtk_cell_area_cell_get_property")]
32    fn cell_get_value(
33        &self,
34        renderer: &impl IsA<CellRenderer>,
35        property_name: impl IntoGStr,
36    ) -> glib::Value {
37        unsafe {
38            property_name.run_with_gstr(|property_name| {
39                let cell_class = glib::Class::<CellArea>::from_type(Self::static_type()).unwrap();
40                let pspec: Option<glib::ParamSpec> =
41                    from_glib_none(ffi::gtk_cell_area_class_find_cell_property(
42                        cell_class.as_ref() as *const _ as *mut ffi::GtkCellAreaClass,
43                        property_name.as_ptr(),
44                    ));
45                let pspec = pspec.unwrap_or_else(|| {
46                    panic!("The CellArea property {property_name} doesn't exists")
47                });
48                let mut value = glib::Value::from_type(pspec.value_type());
49                ffi::gtk_cell_area_cell_get_property(
50                    self.as_ref().to_glib_none().0,
51                    renderer.as_ref().to_glib_none().0,
52                    property_name.as_ptr(),
53                    value.to_glib_none_mut().0,
54                );
55                value
56            })
57        }
58    }
59
60    // rustdoc-stripper-ignore-next
61    /// Similar to [`Self::cell_get_value`] but panics if the value is of a
62    /// different type.
63    #[doc(alias = "gtk_cell_area_cell_get_valist")]
64    #[doc(alias = "gtk_cell_area_cell_get_property")]
65    fn cell_get<V: for<'b> FromValue<'b> + 'static>(
66        &self,
67        renderer: &impl IsA<CellRenderer>,
68        property_name: impl IntoGStr,
69    ) -> V {
70        let value = self.cell_get_value(renderer, property_name);
71        value
72            .get_owned::<V>()
73            .expect("Failed to get value of renderer")
74    }
75
76    #[doc(alias = "gtk_cell_area_cell_set_valist")]
77    #[doc(alias = "gtk_cell_area_cell_set_property")]
78    fn cell_set(
79        &self,
80        renderer: &impl IsA<CellRenderer>,
81        property_name: impl IntoGStr,
82        value: impl Into<Value>,
83    ) {
84        unsafe {
85            property_name.run_with_gstr(|property_name| {
86                let cell_class = glib::Class::<CellArea>::from_type(Self::static_type()).unwrap();
87                let pspec: Option<glib::ParamSpec> =
88                    from_glib_none(ffi::gtk_cell_area_class_find_cell_property(
89                        cell_class.as_ref() as *const _ as *mut ffi::GtkCellAreaClass,
90                        property_name.as_ptr(),
91                    ));
92                let pspec = pspec.unwrap_or_else(|| {
93                    panic!("The CellArea property {property_name} doesn't exists")
94                });
95
96                let value = value.into();
97                assert!(
98                    pspec.value_type().is_a(value.type_()),
99                    "The CellArea property's value is of wrong type. Expected '{}' but got '{}'",
100                    pspec.value_type(),
101                    value.type_()
102                );
103
104                ffi::gtk_cell_area_cell_set_property(
105                    self.as_ref().to_glib_none().0,
106                    renderer.as_ref().to_glib_none().0,
107                    property_name.as_ptr(),
108                    value.to_glib_none().0,
109                );
110            })
111        }
112    }
113}
114
115impl<O: IsA<CellArea>> CellAreaExtManual for O {}
